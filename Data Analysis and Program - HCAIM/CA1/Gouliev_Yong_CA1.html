State and carry out one 2-sample hypothesis test that involves categorical variable(s) (use proportions)

# We test to see if the proportion of female customers is significantly different from the proportion of male customers

male_customers = data[data['gender'] == 'Male']
female_customers = data[data['gender'] == 'Female']

# the proportions of male and female customers
p_male = len(male_customers) / len(data)
p_female = len(female_customers) / len(data)

# standard error of the difference in proportions
se_diff = np.sqrt(p_male*(1-p_male)/len(data) + p_female*(1-p_female)/len(data))

# z-score
z_score = (p_female - p_male) / se_diff

# p-value using a 2-sample hypothesis test
p_value = norm.sf(abs(z_score))*2

#  significance level
sig_level = 0.05

# results
if p_value < sig_level:
    print('The p-value', p_value, 'is less than'. sig_level, 'so we reject the null hypothesis')
else:
    print('The p-value', p_value, 'is greater than', sig_level, 'so we do not reject the null hypothesis')
